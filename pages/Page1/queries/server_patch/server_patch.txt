{{
  (() => {
    const payload = {
      name: PatchServer_Name.text,
      provider: PatchServer_Provider.text,
      status: PatchServer_Status.selectedOptionValue,
      setup_date: PatchServer_SetupDate.selectedDate
        ? moment(PatchServer_SetupDate.selectedDate).local().format("YYYY-MM-DD")
        : null,
      access_email: PatchServer_AccessEmail.text,
      hostname_domain: PatchServer_Hostname.selectedOptionValue
    };

    // remove keys with null, undefined, or empty string
    Object.keys(payload).forEach(k => {
      if (
        payload[k] === "" ||
        payload[k] === null ||
        payload[k] === undefined
      ) {
        delete payload[k];
      }
    });

    return payload;
  })()
}}
