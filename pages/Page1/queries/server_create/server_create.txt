{{
  (() => {
    const payload = {
      // optional
      name: CreateServer_Name.text,
      provider: CreateServer_Provider.text,
      access_email: CreateServer_AccessEmail.text,

      // required
      name_server_1: CreateServer_NS1.text,
      name_server_2: CreateServer_NS2.text,
      ip_address: CreateServer_Ip.text,
    };

    // Clean only optional keys so required ones are always sent
    const optionalKeys = ["name", "provider", "access_email"];

    optionalKeys.forEach((k) => {
      if (
        payload[k] === "" ||
        payload[k] === null ||
        payload[k] === undefined
      ) {
        delete payload[k];
      }
    });

    return payload;
  })()
}}
