{{
  (() => {
    const payload = {
      // required
      name: CreateDomain_Name.text,
			// optional
      domain_provider: CreateDomain_Provider.text,
      access_email: CreateDomain_AccessEmail.text,     
      client: CreateDomain_Client.selectedOptionValue,
			usage: CreateDomain_Usage.selectedOptionValue,
			status: CreateDomain_Status.selectedOptionValue,
			website: CreateDomain_Website.selectedOptionValue,
			created_at: CreateDomain_CreatedAt.selectedDate
        ? moment(CreateDomain_CreatedAt.selectedDate).local().format("YYYY-MM-DD")
        : null
    };

    // Clean only optional keys so required ones are always sent
    const optionalKeys = ["client", "provider", "access_email", "created_at", "status", "website", "usage"];

    optionalKeys.forEach((k) => {
      if (
        payload[k] === "" ||
        payload[k] === null ||
        payload[k] === undefined
      ) {
        delete payload[k];
      }
    });

    return payload;
  })()
}}
