{{
  (() => {
    const payload = {
      // optional rename
      address: PatchMailbox_NewName.text
  ? `${PatchMailbox_NewName.text}@${PatchMailbox_Domain.selectedOptionLabel}`.toLowerCase()
  : undefined,
      role: PatchMailbox_Role.selectedOptionValue,
      status: PatchMailbox_Status.selectedOptionValue,
      warmup_date: PatchMailbox_WarmupDate.selectedDate,
      decommissioned_at: PatchMailbox_DecommissionedAt.selectedDate,
			auto_reply: PatchMailbox_AutoReply.selectedOptionValue,
			url: PatchMailBox_Url.text,
			password: PatchMailBox_Password.text
    };

    // remove keys with null, undefined, or empty string
    Object.keys(payload).forEach(k => {
      if (
        payload[k] === "" ||
        payload[k] === null ||
        payload[k] === undefined
      ) {
        delete payload[k];
      }
    });

    return payload;
  })()
}}
