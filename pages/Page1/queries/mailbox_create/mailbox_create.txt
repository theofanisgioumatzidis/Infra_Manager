{{
  (() => {
    const payload = {
      // required
      address: `${CreateMailbox_Name.text}@${CreateMailbox_Domain.selectedOptionLabel}`.toLowerCase(),
      domain: CreateMailbox_Domain.selectedOptionValue,
			password: CreateMailbox_Password.text,
      role: CreateMailbox_Role.selectedOptionValue,    
			// optional
      status: CreateMailbox_Status.selectedOptionValue,
			warmup_date: CreateMailbox_WarmupDate.selectedDate
        ? moment(CreateMailbox_WarmupDate.selectedDate).local().format("YYYY-MM-DD")
        : null
    };

    // Clean only optional keys so required ones are always sent
    const optionalKeys = ["status", "warmup_date"];

    optionalKeys.forEach((k) => {
      if (
        payload[k] === "" ||
        payload[k] === null ||
        payload[k] === undefined
      ) {
        delete payload[k];
      }
    });

    return payload;
  })()
}}
